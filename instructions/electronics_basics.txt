const instructionData = require('./instructionData');

const displayInstruction = (instructionId) => {
    const instruction = instructionData.find(instruction => instruction.id === instructionId);
    if (!instruction) {
        console.error('error', 'Instruction not found');
        return;
    }
    console.log(instruction.content);
}
